HARSPOOL ALGORITHM
#include<iostream>
using namespace std;
int str_length (char str[])
{
      int i=0;
      while(str[i]!='\0')
      {
         i++;
      }
      return i;
}
void shift_table(char pattern[],int m ,int table[])
{
      for(int i=0;i<256;i++)
      {
         table[i]=m;
      }
      for(int i=0;i<m-1;i++)
      {
         table[(int) pattern[i]]=m-1-i;
      }
}
int horspool (char text[],char pattern[])
{
      int m=str_length(pattern);
      int n=str_length(text);
      int table[256];
      shift_table(pattern,m,table);
      int i=m-1;
      while(i<n)
      {
         int k=0;
         while(k<m && pattern[m-1-k]==text[i-k])
         {
            k++;
         }
         if(k==m)
         {
            return i-m+1;
         }
         else
         {
            i=i+table[(int) text[i]];
         }
      }
      return -1;
}
int main()
{
      char text[200],pattern[100];
      cout<<"Enter the text";
      cin.getline(text,200);
      cout<<"Enter the pattern";
      cin.getline(pattern,100);
      int pos=horspool(text,pattern);
      if(pos>=0)
      {
         cout<<"Pattern found at position"<<pos;
         cout<<endl;
      }
      else
      {
         cout<<"Pattern not found"<<endl;
      }
      return 0;
}
